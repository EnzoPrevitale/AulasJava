public class Main {
	public static void main(String[] args) {
		for(int i = 1; i <= 10; i++) {
			System.out.println("Tabuada do " + i);
			for(int j = 1; j <= 10; j++) {
				System.out.println(i + "X" + j + "=" + i*j);
			}
		}
	}
}

public class Main {
	public static void main(String[] args) {
		int i = 1;
		while(i < 10) {
			int j = 1;
			System.out.println("Tabuada do"+i);
			while(j < 10) {
				System.out.println(i + "X" + j + "=" + i*j);
				j++;
			}
			i++;
		}
	}
}
